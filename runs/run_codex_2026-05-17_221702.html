+async function main() {
+  const args = process.argv.slice(2);
+  const outIndex = args.indexOf(&quot;--out&quot;);
+  if (outIndex === -1 || !args[outIndex + 1]) {
+    usage();
+    process.exit(1);
+  }
+
+  const sourceInput = args.find((arg, index) =&gt; !arg.startsWith(&quot;--&quot;) &amp;&amp; index !== outIndex + 1);
+  const slug = slugFromInput(sourceInput);
+  const [detailsMd, reviewsMd, supportMd] = await Promise.all([
+    fetchText(`https://wordpress.org/plugins/${slug}/?output_format=md`),
+    fetchText(`https://wordpress.org/support/plugin/${slug}/reviews/?output_format=md`),
+    fetchText(`https://wordpress.org/support/plugin/${slug}/?output_format=md`),
+  ]);
+
+  const details = parseDetails(detailsMd, slug);
+  const reviews = parseReviews(reviewsMd);
+  const support = parseSupport(supportMd);
+  const themes = scoreThemes(details, reviews, support);
+
+  await fs.writeFile(args[outIndex + 1], buildReport({ slug, details, reviews, support, themes }));
+  console.log(`Wrote ${args[outIndex + 1]} for ${slug}`);
+}
+
+main().catch((error) =&gt; {
+  console.error(error.message);
+  process.exit(1);
+});
